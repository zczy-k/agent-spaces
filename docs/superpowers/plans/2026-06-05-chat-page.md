// packages/server/src/storage/chat-store.ts
import { existsSync, readdirSync, unlinkSync } from 'node:fs';
import { join } from 'node:path';
import { ensureDir, getDataDir, readJsonFile, writeJsonFile, deleteFile } from './json-store.js';

export interface ChatAgent {
  id: string;
  name: string;
  avatar?: string;
  description?: string;
  systemPrompt?: string;
  provider: string;
  model: string;
  apiKey: string;
  baseURL?: string;
  createdAt: string;
  updatedAt: string;
}

export interface ChatMessage {
  id: string;
  agentId: string;
  role: 'user' | 'agent';
  content: string;
  timestamp: string;
  thinking?: string;
  usage?: {
    inputTokens?: number;
    outputTokens?: number;
    totalTokens?: number;
  };
}

function chatDir(): string {
  return join(getDataDir(), 'chat');
}

function agentsFile(): string {
  return join(chatDir(), 'agents.json');
}

function messageHistoryDir(): string {
  return join(chatDir(), 'message_history');
}

// --- Agents ---

export function listAgents(): ChatAgent[] {
  return readJsonFile<ChatAgent[]>(agentsFile()) ?? [];
}

export function saveAgents(agents: ChatAgent[]): void {
  ensureDir(chatDir());
  writeJsonFile(agentsFile(), agents);
}

export function findAgent(id: string): ChatAgent | undefined {
  return listAgents().find((a) => a.id === id);
}

// --- Messages ---

export function listMessages(agentId: string, limit = 50, before?: string): ChatMessage[] {
  const dir = messageHistoryDir();
  if (!existsSync(dir)) return [];

  const files = readdirSync(dir)
    .filter((f) => f.endsWith('.json'))
    .sort()
    .reverse(); // newest first by filename (timestamp-based uuid sort is rough; we'll filter+sort below)

  const messages: ChatMessage[] = [];
  let hitCursor = !before; // if no cursor, start collecting immediately

  for (const file of files) {
    const msg = readJsonFile<ChatMessage>(join(dir, file));
    if (!msg || msg.agentId !== agentId) continue;
    if (before && msg.id === before) { hitCursor = true; continue; }
    if (hitCursor) messages.push(msg);
    if (messages.length >= limit) break;
  }

  // Sort ascending (oldest first) for display
  return messages.reverse();
}

export function saveMessage(msg: ChatMessage): void {
  const dir = messageHistoryDir();
  ensureDir(dir);
  writeJsonFile(join(dir, `${msg.id}.json`), msg);
}

export function deleteMessagesByAgent(agentId: string): void {
  const dir = messageHistoryDir();
  if (!existsSync(dir)) return;

  const files = readdirSync(dir).filter((f) => f.endsWith('.json'));
  for (const file of files) {
    const msg = readJsonFile<ChatMessage>(join(dir, file));
    if (msg && msg.agentId === agentId) {
      deleteFile(join(dir, file));
    }
  }
}

export function getRecentMessages(agentId: string, limit = 20): ChatMessage[] {
  const dir = messageHistoryDir();
  if (!existsSync(dir)) return [];

  const files = readdirSync(dir)
    .filter((f) => f.endsWith('.json'))
    .sort()
    .reverse();

  const messages: ChatMessage[] = [];
  for (const file of files) {
    const msg = readJsonFile<ChatMessage>(join(dir, file));
    if (msg && msg.agentId === agentId) {
      messages.push(msg);
      if (messages.length >= limit) break;
    }
  }

  return messages.reverse();
}
```

// packages/server/src/services/chat.ts
import { randomUUID } from 'node:crypto';
import * as store from '../storage/chat-store.js';
import type { ChatAgent, ChatMessage } from '../storage/chat-store.js';

// --- Agent CRUD ---

export function listAgents(): ChatAgent[] {
  return store.listAgents();
}

export function createAgent(data: Omit<ChatAgent, 'id' | 'createdAt' | 'updatedAt'>): ChatAgent {
  const now = new Date().toISOString();
  const agent: ChatAgent = {
    ...data,
    id: randomUUID(),
    createdAt: now,
    updatedAt: now,
  };
  const agents = store.listAgents();
  agents.push(agent);
  store.saveAgents(agents);
  return agent;
}

export function updateAgent(id: string, data: Partial<Omit<ChatAgent, 'id' | 'createdAt'>>): ChatAgent | null {
  const agents = store.listAgents();
  const index = agents.findIndex((a) => a.id === id);
  if (index === -1) return null;
  agents[index] = { ...agents[index], ...data, updatedAt: new Date().toISOString() };
  store.saveAgents(agents);
  return agents[index];
}

export function deleteAgent(id: string): boolean {
  const agents = store.listAgents();
  const index = agents.findIndex((a) => a.id === id);
  if (index === -1) return false;
  agents.splice(index, 1);
  store.saveAgents(agents);
  store.deleteMessagesByAgent(id);
  return true;
}

// --- Message CRUD ---

export function listMessages(agentId: string, limit?: number, before?: string): ChatMessage[] {
  return store.listMessages(agentId, limit, before);
}

export function saveMessage(msg: Omit<ChatMessage, 'id' | 'timestamp'>): ChatMessage {
  const message: ChatMessage = {
    ...msg,
    id: randomUUID(),
    timestamp: new Date().toISOString(),
  };
  store.saveMessage(message);
  return message;
}

export function clearMessages(agentId: string): void {
  store.deleteMessagesByAgent(agentId);
}

export function getRecentMessages(agentId: string, limit = 20): ChatMessage[] {
  return store.getRecentMessages(agentId, limit);
}

export function findAgent(id: string): ChatAgent | undefined {
  return store.findAgent(id);
}

// packages/server/src/routes/chat.ts
import { Router } from 'express';
import * as svc from '../services/chat.js';

const router = Router();

// List agents
router.get('/agents', (_req, res) => {
  try {
    res.json(svc.listAgents());
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});

// Create agent
router.post('/agents', (req, res) => {
  try {
    const { name, provider, model, apiKey } = req.body;
    if (!name || !provider || !model || !apiKey) {
      res.status(400).json({ error: 'name, provider, model, and apiKey are required' });
      return;
    }
    const agent = svc.createAgent(req.body);
    res.status(201).json(agent);
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});

// Update agent
router.put('/agents/:id', (req, res) => {
  try {
    const agent = svc.updateAgent(req.params.id, req.body);
    if (!agent) {
      res.status(404).json({ error: 'Agent not found' });
      return;
    }
    res.json(agent);
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});

// Delete agent
router.delete('/agents/:id', (req, res) => {
  try {
    const ok = svc.deleteAgent(req.params.id);
    if (!ok) {
      res.status(404).json({ error: 'Agent not found' });
      return;
    }
    res.status(204).end();
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});

// List messages
router.get('/agents/:id/messages', (req, res) => {
  try {
    const limit = parseInt(req.query.limit as string) || 50;
    const before = req.query.before as string | undefined;
    res.json(svc.listMessages(req.params.id, limit, before));
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});

// Clear messages
router.delete('/agents/:id/messages', (req, res) => {
  try {
    svc.clearMessages(req.params.id);
    res.status(204).end();
  } catch (error) {
    res.status(500).json({ error: (error as Error).message });
  }
});

export default router;

// packages/server/src/ws/chat-handler.ts
import type { WebSocket } from 'ws';
import { broadcastToWorkspace } from './connection-manager.js';
import { registerHandler } from './handler.js';
import * as chatService from '../services/chat.js';
import type { ChatAgent, ChatMessage } from '../storage/chat-store.js';
import { LangChainRuntime } from '../adapters/langchain-runtime.js';
import type { AgentRuntimeConfig, AgentRuntimeEvent } from '../adapters/agent-runtime-types.js';

const activeRuns = new Map<string, LangChainRuntime>(); // agentId -> runtime

function broadcast(workspaceId: string, event: string, data: unknown) {
  broadcastToWorkspace(workspaceId, event, data);
}

export function registerChatHandlers(): void {
  registerHandler('chat.message', (_ws, workspaceId, data) => {
    const { agentId, content } = data as { agentId: string; content: string };
    if (!agentId || !content) return;

    const agent = chatService.findAgent(agentId);
    if (!agent) {
      broadcast(workspaceId, 'chat.agent.error', { agentId, error: 'Agent not found' });
      return;
    }

    // Save user message
    const userMsg = chatService.saveMessage({ agentId, role: 'user', content });
    broadcast(workspaceId, 'chat.message.saved', userMsg);

    // Build LangChain runtime config
    const config: AgentRuntimeConfig = {
      kind: 'langchain',
      provider: agent.provider,
      model: agent.model,
      apiKey: agent.apiKey,
      baseURL: agent.baseURL,
    };

    const runtime = new LangChainRuntime(config);
    activeRuns.set(agentId, runtime);

    // Load recent messages as conversation history
    const recentMessages = chatService.getRecentMessages(agentId, 20);
    const historyPrompt = recentMessages
      .map((m) => `${m.role === 'user' ? 'User' : 'Assistant'}: ${m.content}`)
      .join('\n\n');

    const prompt = historyPrompt
      ? `${historyPrompt}\n\nUser: ${content}\nAssistant:`
      : content;

    // Execute with streaming events
    const runPromise = runtime.execute(prompt, process.cwd(), {
      systemPrompt: agent.systemPrompt,
      onEvent: (event: AgentRuntimeEvent) => {
        switch (event.type) {
          case 'text':
            broadcast(workspaceId, 'chat.agent.output', { agentId, chunk: event.text });
            break;
          case 'thinking':
            broadcast(workspaceId, 'chat.agent.thinking', { agentId, chunk: event.text });
            break;
          case 'tool_use':
            broadcast(workspaceId, 'chat.agent.tool_use', {
              agentId,
              name: event.name,
              input: event.input,
            });
            break;
          case 'tool_result':
            broadcast(workspaceId, 'chat.agent.tool_result', {
              agentId,
              name: event.toolUseId,
              result: event.result,
            });
            break;
        }
      },
    });

    runPromise
      .then((result) => {
        const agentContent = result.output.filter((line) => !line.startsWith('Tool:')).join('\n') || result.summary;
        const agentMsg = chatService.saveMessage({
          agentId,
          role: 'agent',
          content: agentContent,
          thinking: undefined,
          usage: result.usage,
        });
        broadcast(workspaceId, 'chat.agent.completed', { agentId, message: agentMsg });
      })
      .catch((err) => {
        broadcast(workspaceId, 'chat.agent.error', {
          agentId,
          error: err instanceof Error ? err.message : String(err),
        });
      })
      .finally(() => {
        activeRuns.delete(agentId);
      });
  });

  registerHandler('chat.stop', (_ws, _workspaceId, data) => {
    const { agentId } = data as { agentId: string };
    if (!agentId) return;
    const runtime = activeRuns.get(agentId);
    if (runtime) {
      runtime.stop();
      activeRuns.delete(agentId);
    }
  });
}

// packages/web/src/stores/chat.ts
"use client";
import { create } from 'zustand';
import { sdk } from '@/lib/sdk';
import type { ChatAgent, ChatMessage } from '@agent-spaces/sdk';

interface ChatStore {
  agents: ChatAgent[];
  activeAgentId: string | null;
  messages: Record<string, ChatMessage[]>;
  sending: Record<string, boolean>;

  loadAgents: () => Promise<void>;
  createAgent: (data: Omit<ChatAgent, 'id' | 'createdAt' | 'updatedAt'>) => Promise<void>;
  updateAgent: (id: string, data: Partial<ChatAgent>) => Promise<void>;
  deleteAgent: (id: string) => Promise<void>;
  selectAgent: (id: string) => void;
  loadMessages: (agentId: string) => Promise<void>;
  sendMessage: (agentId: string, content: string) => void;
  stopAgent: (agentId: string) => void;

  // WS event handlers (called from WS listener)
  onMessageSaved: (msg: ChatMessage) => void;
  onAgentOutput: (agentId: string, chunk: string) => void;
  onAgentThinking: (agentId: string, chunk: string) => void;
  onAgentCompleted: (agentId: string, message: ChatMessage) => void;
  onAgentError: (agentId: string, error: string) => void;
}

export const useChatStore = create<ChatStore>((set, get) => ({
  agents: [],
  activeAgentId: null,
  messages: {},
  sending: {},

  loadAgents: async () => {
    try {
      const agents = await sdk.chat.listAgents();
      set({ agents });
    } catch { /* ignore */ }
  },

  createAgent: async (data) => {
    const agent = await sdk.chat.createAgent(data);
    set((s) => ({ agents: [...s.agents, agent] }));
  },

  updateAgent: async (id, data) => {
    const updated = await sdk.chat.updateAgent(id, data);
    set((s) => ({
      agents: s.agents.map((a) => (a.id === id ? updated : a)),
    }));
  },

  deleteAgent: async (id) => {
    await sdk.chat.deleteAgent(id);
    set((s) => {
      const { [id]: _removed, ...restMessages } = s.messages;
      const { [id]: _sending, ...restSending } = s.sending;
      return {
        agents: s.agents.filter((a) => a.id !== id),
        messages: restMessages,
        sending: restSending,
        activeAgentId: s.activeAgentId === id ? null : s.activeAgentId,
      };
    });
  },

  selectAgent: (id) => {
    set({ activeAgentId: id });
    // Load messages if not cached
    const { messages } = get();
    if (!messages[id]) {
      get().loadMessages(id);
    }
  },

  loadMessages: async (agentId) => {
    try {
      const msgs = await sdk.chat.listMessages(agentId);
      set((s) => ({ messages: { ...s.messages, [agentId]: msgs } }));
    } catch { /* ignore */ }
  },

  sendMessage: (agentId, content) => {
    // This will be called from the WS connection, which sends the event
    // For now, mark as sending and emit the WS event
    set((s) => ({ sending: { ...s.sending, [agentId]: true } }));
  },

  stopAgent: (agentId) => {
    // Will emit WS event via the connection
    set((s) => ({ sending: { ...s.sending, [agentId]: false } }));
  },

  // WS event handlers
  onMessageSaved: (msg) => {
    set((s) => ({
      messages: {
        ...s.messages,
        [msg.agentId]: [...(s.messages[msg.agentId] ?? []), msg],
      },
    }));
  },

  onAgentOutput: (agentId, _chunk) => {
    // Accumulate output chunks into the last agent message
    // For simplicity, we'll append to the last message content on completed
  },

  onAgentThinking: (_agentId, _chunk) => {
    // Similar accumulation for thinking
  },

  onAgentCompleted: (agentId, message) => {
    set((s) => ({
      messages: {
        ...s.messages,
        [agentId]: [...(s.messages[agentId] ?? []), message],
      },
      sending: { ...s.sending, [agentId]: false },
    }));
  },

  onAgentError: (agentId, _error) => {
    set((s) => ({ sending: { ...s.sending, [agentId]: false } }));
  },
}));

function extractThinkingContent(content: string): { thinking: string | null; message: string } {
  const match = content.match(/^<think\s*>([\s\S]*?)<\/think>\s*([\s\S]*)$/);
  if (match) return { thinking: match[1].trim(), message: match[2].trim() };
  return { thinking: null, message: content };

function formatTime(ts: string) {

}

import { useRef, useEffect, useState } from "react";

interface InlineChatPanelProps {
  agentName: string;
  agentAvatar?: string;
  messages: ChatMessage[];
  sending: boolean;
  input: string;
  onInputChange: (value: string) => void;
  onSend: () => void;
  onStop: () => void;
}

export function InlineChatPanel({
  agentName,
  agentAvatar,
  messages,
  sending,
  input,
  onInputChange,
  onSend,
  onStop,
}: InlineChatPanelProps) {
  const listRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!listRef.current) return;
    listRef.current.scrollTo({ top: listRef.current.scrollHeight, behavior: "smooth" });
  }, [messages, sending]);

        </Avatar>

          <span className="text-xs text-muted-foreground">
            {sending ? "typing..." : "online"}
          </span>
        </div>
      </div>

      {/* Messages */}
      <div ref={listRef} className="flex-1 overflow-y-auto p-4">
        <div className="flex flex-col gap-3">
          {messages.length === 0 && !sending && (
            <p className="py-12 text-center text-xs text-muted-foreground">

          )}
        </div>
      </div>

      {/* Input */}
      <div className="border-t p-3">

            )}
          </Button>
        </form>
      </div>
    </div>
  );
}

// packages/web/src/hooks/use-chat-ws.ts
"use client";

import { useEffect, useRef } from "react";
import { useChatStore } from "@/stores/chat";
import { getWS } from "@/lib/ws";

export function useChatWS(workspaceId: string | null) {
  const { onMessageSaved, onAgentCompleted, onAgentError } = useChatStore();
  const registered = useRef(false);

  useEffect(() => {
    if (!workspaceId || registered.current) return;
    registered.current = true;

    const ws = getWS(workspaceId);
    if (!ws) return;

    ws.on("chat.message.saved", (data) => {
      onMessageSaved(data as any);
    });

    ws.on("chat.agent.completed", (data: any) => {
      onAgentCompleted(data.agentId, data.message);
    });

    ws.on("chat.agent.error", (data: any) => {
      onAgentError(data.agentId, data.error);
    });

    return () => {
      ws.off("chat.message.saved");
      ws.off("chat.agent.completed");
      ws.off("chat.agent.error");
    };
  }, [workspaceId, onMessageSaved, onAgentCompleted, onAgentError]);
}
```

Note: The actual WS emit for `chat.message` from the frontend will need to go through `ws.send({ event: 'chat.message', data: { agentId, content } })`. This depends on how the existing `WorkspaceWS` class exposes send. Check `packages/web/src/lib/ws.ts` for the `emit` method pattern.
